abstract RGBA(Int)

package lime.math

from Int, UInt, to Int, UInt

@:allow(lime.math)@:transitive

A utility for storing, accessing and converting colors in an RGBA (red, green, blue, alpha) color format.

var color:RGBA = 0x883300FF;
trace (color.r); // 0x88
trace (color.g); // 0x33
trace (color.b); // 0x00
trace (color.a); // 0xFF

var convert:ARGB = color; // 0xFF883300
