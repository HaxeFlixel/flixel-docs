abstract ARGB(Int)

package lime.math

from Int, UInt, to Int, UInt

@:transitive

A utility for storing, accessing and converting colors in an ARGB (alpha, red, green, blue) color format.

var color:ARGB = 0xFF883300;
trace (color.a); // 0xFF
trace (color.r); // 0x88
trace (color.g); // 0x33
trace (color.b); // 0x00

var convert:BGRA = color; // 0x003388FF
